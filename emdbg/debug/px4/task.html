 72    """
 73    NuttX task
 74    """
 75    _STACK_COLOR = 0xdeadbeef
 76    _FILE_DESCRIPTORS_PER_BLOCK = 6 # TODO query from ELF
 77
 78    @dataclass
 79    class Load:
 80        total: int
 81        """Total task runtime in µs"""
 82        interval: int
 83        """Interval task runtime in µs since last sample"""
 84        delta: int
 85        """Task runtime in µs within the interval"""
 86
 87        @property
 88        def relative(self) -> float:
 89            """Relative runtime within the interval"""
 90            return self.delta / self.interval if self.interval else 0

 92    def __init__(self, gdb, tcb_ptr: "gdb.Value"):
 93        super().__init__(gdb)
 94        self._tcb = tcb_ptr
 95        self._system_load = system_load(self._gdb)
 96        self.pid = self._tcb["pid"]
 97        self.init_priority = int(self._tcb["init_priority"])
 98        self.stack_limit = int(self._tcb["adj_stack_size"])
 99        self.stack_ptr = self._tcb["stack_base_ptr"]
100        self._is_running_switched = None
101
102    @cached_property
103    def name(self) -> str:
104        """Name of the task"""
105        try:
106            return self._tcb["name"].string()
107        except:
108            return "?"
109
110    @cached_property
111    def sched_priority(self) -> int:
112        """The scheduled priority of the task"""
113        return int(self._tcb["sched_priority"])
114
115    @cached_property
116    def _statenames(self):
117        return self._gdb.types.make_enum_dict(self._gdb.lookup_type("enum tstate_e"))
118
119    @cached_property
120    def state(self) -> str:
121        """Task state name"""
122        for name, value in self._statenames.items():
123            if value == self._tcb["task_state"]:
124                return name
125        return "UNKNOWN"
126
127    @cached_property
128    def short_state(self) -> str:
129        """The short name of the task state"""
130        mapping = {
131            "TASK_PENDING": "PEND", "TASK_READYTORUN": "READY", "TASK_RUNNING": "RUN",
132            "WAIT_SEM": "w:sem", "WAIT_SIG": "w:sig"}
133        return mapping.get(self.state.replace("TSTATE_", ""), "???")
134
135    def _is_state_in(self, *states: list[str]) -> bool:
136        states = {int(self._statenames[s]) for s in states}
137        if int(self._tcb["task_state"]) in states:
138            return True
139        return False
140
141    @cached_property
142    def is_waiting(self) -> bool:
143        """The task is waiting for a semaphore or signal"""
144        if self._is_state_in("TSTATE_WAIT_SEM", "TSTATE_WAIT_SIG"):
145            return True
146
147    @cached_property
148    def is_runnable(self) -> bool:
149        """The task is pending, ready to run, or running"""
150        return self._is_state_in("TSTATE_TASK_PENDING",
151                                 "TSTATE_TASK_READYTORUN",
152                                 "TSTATE_TASK_RUNNING")
153
154    @cached_property
155    def stack_used(self) -> int:
156        """The amount of stack used by the thread in bytes"""
157        if not self.stack_ptr: return 0
158        stack_u32p = self.stack_ptr.cast(self._gdb.lookup_type("unsigned int").pointer())
159        # Stack grows from top to bottom, we do a binary search for the
160        # 0xdeadbeef value from the bottom upwards
161        watermark = utils.binary_search_last(stack_u32p, self._STACK_COLOR, hi=self.stack_limit // 4) + 1
162        # validate the binary search (does not seem necessary)
163        # for ii in range(0, watermark):
164        #     if stack_u32p[ii] != self._STACK_COLOR:
165        #         print(f"{self.name}: Correcting stack size from {watermark * 4} to {ii * 4}!")
166        #         return ii * 4
167        return self.stack_limit - watermark * 4
168
169    @cached_property
170    def waiting_for(self) -> str:
171        """
172        What the task is waiting for. If its a semaphore, return an object,
173        otherwise a string.
174        """
175        if self._is_state_in("TSTATE_WAIT_SEM"):
176            from .semaphore import Semaphore
177            sem = self._tcb['waitsem']
178            ostr = f"{int(sem):#08x} "
179            if descr := self.description_at(sem): ostr += f"<{descr}> "
180            ostr += Semaphore(self._gdb, sem).to_string()
181            return ostr
182        if self._is_state_in("TSTATE_WAIT_SIG"):
183            return "signal"
184        return ""
185
186    @cached_property
187    def files(self) -> list["gdb.Value"]:
188        """The list of inode pointers the task holds"""
189        filelist = self._tcb["group"]["tg_filelist"]
190        rows = filelist["fl_rows"]
191        files = filelist["fl_files"]
192        result = []
193        for ri in range(rows):
194            for ci in range(self._FILE_DESCRIPTORS_PER_BLOCK):
195                file = files[ri][ci]
196                if inode := file["f_inode"]:
197                    result.append(file)
198        return result
199
200    @cached_property
201    def location(self) -> "gdb.Block":
202        """The block the task is currently executing"""
203        if self.is_current_task:
204            pc = self.read_register("pc")
205        else:
206            pc = self._tcb["xcp"]["regs"][32]
207        block = self.block(pc)
208        while block and not block.function:
209            block = block.superblock
210        return block.function if block else int(pc)
211
212    def switch_to(self) -> bool:
213        """Switch to this task by writing the register file"""
214        if self.is_current_task:
215            return False
216        regs = {name: self._tcb["xcp"]["regs"][offset]
217                for name, offset in _XCP_REGS_MAP.items()}
218        self.write_registers(regs)
219        self._is_running_switched = True
220        return True
221
222    def switch_from(self) -> dict[str, int] | None:
223        """Switch to this task by writing the register file"""
224        if not self.is_current_task:
225            return None
226        self._is_running_switched = False
227        if self.short_state == "RUN":
228            self._is_running_switched = None
229        return self.registers
230
231    @cached_property
232    def is_current_task(self) -> bool:
233        """If the task is currently running"""
234        if self._is_running_switched is not None:
235            return self._is_running_switched
236        return self.short_state == "RUN"
237
238    @cached_property
239    def load(self) -> Load:
240        """The task load based on the system load monitor"""
241        if self._system_load is None: return self.Load(0, 0, 0)
242        _, interval, sl = self._system_load.sample
243        total, delta = sl.get(int(self._tcb), (0,0))
244        return self.Load(total, interval, delta)
245
246    def __repr__(self) -> str:
247        return f"Task({self.name}, {self.pid})"
248
249    def __str__(self) -> str:
250        ostr = self.__repr__() + f": state={self.state}, "
251        ostr += f"prio={self.sched_priority}({self.init_priority}), "
252        ostr += f"stack={self.stack_used}/{self.stack_limit}"
253        if waiting := self.waiting_for:
254            ostr += f", waiting={waiting}"
255        return ostr

258def all_tasks(gdb) -> list[Task]:
259    """Return a list of all tasks"""
260    type_tcb_s = gdb.lookup_type("struct tcb_s").pointer()
261    def _tasks(name):
262        tcbs = []
263        if (task_list := gdb.lookup_global_symbol(name)) is None:
264            return []
265        task_list = task_list.value()
266        current_task = task_list["head"]
267        if current_task:
268            while True:
269                tcbs.append(Task(gdb, current_task.cast(type_tcb_s)))
270                if current_task == task_list["tail"]:
271                    break
272                next_task = current_task["flink"]
273                # if next_task["blink"] == current_task:
274                #   LOGGER.error(f"Task linkage is broken in {tasks}!")
275                #   break
276                current_task = next_task
277        return tcbs
278
279    tcbs  = _tasks("g_pendingtasks")
280    tcbs += _tasks("g_readytorun")
281    tcbs += _tasks("g_waitingforsemaphore")
282    tcbs += _tasks("g_waitingforsignal")
283    tcbs += _tasks("g_inactivetasks")
284    return tcbs

287_RESTORE_REGISTERS = None
288def task_switch(gdb, pid: int) -> bool:
289    """
290    Switch to another task.
291    On initial switch the current register file is saved and can be restored
292    by passing a PID <0.
293
294    :param pid: the PID of the task to switch to, or <0 to restore initial task.
295    :return: Success of switching operation.
296    """
297    global _RESTORE_REGISTERS
298    # Restore registers to original task
299    if pid < 0:
300        if _RESTORE_REGISTERS is not None:
301            Base(gdb).write_registers(_RESTORE_REGISTERS)
302        _RESTORE_REGISTERS = None
303        return True
304
305    # Otherwise find the new pointer
306    tcbs = all_tasks(gdb)
307    if (next_task := next((t for t in tcbs if int(t.pid) == pid), None)) is not None:
308        # Find the currently executing task and save their registers
309        if (current_task := next((t for t in tcbs if t.is_current_task), None)) is not None:
310            regs = current_task.switch_from()
311        else:
312            regs = Base(gdb).registers
313        # We only care about the first register set for restoration
314        if next_task.switch_to():
315            if _RESTORE_REGISTERS is None:
316                _RESTORE_REGISTERS = regs
317            print(f"Switched to task '{next_task.name}' ({pid}).")
318            return True
319
320        print("Task already loaded!")
321        return False
322
323    print(f"Unknown task PID '{pid}'!")
324    return False

327def all_tasks_as_table(gdb, sort_key: str = None, with_stack_usage: bool = True,
328                       with_file_names: bool = True, with_waiting: bool = True) \
329                                    -> tuple[Table, str] | tuple[None, None]:
330    """
331    Return a table of running tasks similar to the NSH top command.
332
333    :param sort_key: optional lambda function to sort the table rows.
334    :param with_stack_usage: compute and show the task stack usage.
335    :param with_file_names: show what files the task has open.
336    :param with_waiting: show what the task is waiting for.
337
338    :return: The task table and additional output. If no tasks are found, return `None`.
339    """
340    table = Table(box=rich.box.MINIMAL_DOUBLE_HEAD)
341    table.add_column("struct tcb_s*", justify="right", no_wrap=True)
342    table.add_column("pid", justify="right", no_wrap=True)
343    table.add_column("Task Name")
344    table.add_column("Location", no_wrap=True)
345    table.add_column("CPU(ms)", justify="right")
346    table.add_column("CPU(%)", justify="right")
347    table.add_column("Stack\nUsage", justify="right")
348    table.add_column("Stack\nAvail", justify="right")
349    table.add_column("Prio", justify="right")
350    table.add_column("Base", justify="right")
351    if with_file_names: table.add_column("Open File Names")
352    else: table.add_column("FDs", justify="right")
353    table.add_column("State")
354    if with_waiting: table.add_column("Waiting For")
355
356    tasks = all_tasks(gdb)
357    if not tasks: return None, None
358    interval_us = tasks[0].load.interval
359    if not interval_us:
360        start, *_ = tasks[0]._system_load.sample
361        total_interval_us = (Device(gdb).uptime - start) or 1
362    total_user_us, total_idle_us = 0, 0
363    rows = []
364    for task in tasks:
365        # Remember the CPU loads for idle task and add it for the other tasks
366        if task.pid == 0: total_idle_us = task.load.delta if interval_us else task.load.total
367        else: total_user_us += task.load.delta if interval_us else task.load.total
368
369        # Find the file names or just the number of file descriptors
370        if with_file_names:
371            file_names = [task.read_string(f["f_inode"]["i_name"]) for f in task.files]
372            file_description = ", ".join(sorted(list(set(file_names))))
373        else:
374            file_description = len(task.files)
375
376        # Add all the values per row
377        relative = task.load.relative if interval_us else task.load.total / total_interval_us
378        stack_overflow = with_stack_usage and task.stack_used >= (task.stack_limit - max(8, task.stack_limit * 0.1))
379        row = [hex(task._tcb), task.pid, task.name,
380               hex(task.location) if isinstance(task.location, int) else task.location.name,
381               task.load.total//1000, f"{(relative * 100):.1f}",
382               Text.assemble((str(task.stack_used) if with_stack_usage else "", "bold red" if stack_overflow else "")),
383               Text.assemble((str(task.stack_limit), "bold" if stack_overflow else "")),
384               Text.assemble((str(task.sched_priority), "bold red" if task.sched_priority > task.init_priority else "")),
385               task.init_priority, file_description, task.short_state]
386        if with_waiting:
387            row.append(task.waiting_for)
388        rows.append(row)
389
390    # Sort the rows by PID by default and format the table
391    for row in sorted(rows, key=lambda l: l[1] if sort_key is None else sort_key):
392        table.add_row(*[r if isinstance(r, Text) else str(r) for r in row],
393                      style="bold blue" if row[11] == "RUN" else None)
394
395    # Add the task information
396    running = sum(1 for t in tasks if t.is_runnable)
397    sleeping = sum(1 for t in tasks if t.is_waiting)
398    output = f"Processes: {len(tasks)} total, {running} running, {sleeping} sleeping\n"
399
400    # Add CPU utilization and guard against division by zero
401    if not interval_us: interval_us = total_interval_us
402    user = 100 * total_user_us / interval_us;
403    idle = 100 * total_idle_us / interval_us;
404    sched = 100 * (interval_us - total_idle_us - total_user_us) / interval_us;
405    output += f"CPU usage: {user:.1f}% tasks, {sched:.1f}% sched, {idle:.1f}% idle\n"
406
407    # Uptime finally
408    output += f"Uptime: {Device(gdb).uptime/1e6:.2f}s total, {interval_us/1e6:.2f}s interval\n"
409    return table, output

412def all_files_as_table(gdb, sort_key: str = None) -> Table | None:
413    """
414    Return a table of open files owned by tasks.
415
416    :param sort_key: optional lambda function to sort the table rows.
417
418    :return: The file table or `None` if no tasks exist.
419    """
420    tasks = all_tasks(gdb)
421    if not tasks: return None
422
423    files = {}
424    file_tasks = defaultdict(set)
425    for task in tasks:
426        # Find the file names or just the number of file descriptors
427        for f in task.files:
428            files[int(f["f_inode"])] = f["f_inode"]
429            file_tasks[int(f["f_inode"])].add(task)
430    # Format the rows
431    rows = []
432    for addr, inode in files.items():
433        rows.append((hex(addr),
434                     hex(inode["i_private"]) if inode["i_private"] else "",
435                     task.read_string(inode["i_name"]),
436                     ", ".join(sorted(t.name for t in file_tasks[addr]))))
437    # sort and add rows
438    table = Table(box=rich.box.MINIMAL_DOUBLE_HEAD)
439    table.add_column("struct inode*", justify="right", no_wrap=True)
440    table.add_column("i_private*", justify="right", no_wrap=True)
441    table.add_column("Name")
442    table.add_column("Tasks")
443    for row in sorted(rows, key=lambda l: l[2] if sort_key is None else sort_key):
444        table.add_row(*row)
445    return table

290    """
291    Switch to another task.
292    On initial switch the current register file is saved and can be restored
293    by passing a PID <0.
294
295    :param pid: the PID of the task to switch to, or <0 to restore initial task.
296    :return: Success of switching operation.
297    """
298    global _RESTORE_REGISTERS
299    # Restore registers to original task
300    if pid < 0:
301        if _RESTORE_REGISTERS is not None:
302            Base(gdb).write_registers(_RESTORE_REGISTERS)
303        _RESTORE_REGISTERS = None
304        return True
305
306    # Otherwise find the new pointer
307    tcbs = all_tasks(gdb)
308    if (next_task := next((t for t in tcbs if int(t.pid) == pid), None)) is not None:
309        # Find the currently executing task and save their registers
310        if (current_task := next((t for t in tcbs if t.is_current_task), None)) is not None:
311            regs = current_task.switch_from()
312        else:
313            regs = Base(gdb).registers
314        # We only care about the first register set for restoration
315        if next_task.switch_to():
316            if _RESTORE_REGISTERS is None:
317                _RESTORE_REGISTERS = regs
318            print(f"Switched to task '{next_task.name}' ({pid}).")
319            return True
320
321        print("Task already loaded!")
322        return False
323
324    print(f"Unknown task PID '{pid}'!")
325    return False

343    table.add_column("pid", justify="right", no_wrap=True)

351    table.add_column("Base", justify="right")
352    if with_file_names: table.add_column("Open File Names")
353    else: table.add_column("FDs", justify="right")
354    table.add_column("State")
355    if with_waiting: table.add_column("Waiting For")

357    tasks = all_tasks(gdb)
358    if not tasks: return None, None

360    if not interval_us:
361        start, *_ = tasks[0]._system_load.sample
362        total_interval_us = (Device(gdb).uptime - start) or 1
363    total_user_us, total_idle_us = 0, 0
364    rows = []
365    for task in tasks:
366        # Remember the CPU loads for idle task and add it for the other tasks
367        if task.pid == 0: total_idle_us = task.load.delta if interval_us else task.load.total
368        else: total_user_us += task.load.delta if interval_us else task.load.total
369
370        # Find the file names or just the number of file descriptors
371        if with_file_names:
372            file_names = [task.read_string(f["f_inode"]["i_name"]) for f in task.files]
373            file_description = ", ".join(sorted(list(set(file_names))))
374        else:
375            file_description = len(task.files)
376
377        # Add all the values per row
378        relative = task.load.relative if interval_us else task.load.total / total_interval_us
379        stack_overflow = with_stack_usage and task.stack_used >= (task.stack_limit - max(8, task.stack_limit * 0.1))
380        row = [hex(task._tcb), task.pid, task.name,

382               task.load.total//1000, f"{(relative * 100):.1f}",

384               Text.assemble((str(task.stack_limit), "bold" if stack_overflow else "")),
385               Text.assemble((str(task.sched_priority), "bold red" if task.sched_priority > task.init_priority else "")),
386               task.init_priority, file_description, task.short_state]

389        rows.append(row)
390

392    for row in sorted(rows, key=lambda l: l[1] if sort_key is None else sort_key):

394                      style="bold blue" if row[11] == "RUN" else None)

397    running = sum(1 for t in tasks if t.is_runnable)
398    sleeping = sum(1 for t in tasks if t.is_waiting)
399    output = f"Processes: {len(tasks)} total, {running} running, {sleeping} sleeping\n"
400
401    # Add CPU utilization and guard against division by zero
402    if not interval_us: interval_us = total_interval_us
403    user = 100 * total_user_us / interval_us;
404    idle = 100 * total_idle_us / interval_us;
